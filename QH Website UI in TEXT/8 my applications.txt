        // Filter functionality
        function filterApplications() {
            const statusFilter = document.getElementById('statusFilter').value;
            const dateFrom = document.getElementById('dateFrom').value;
            const dateTo = document.getElementById('dateTo').value;
            const amountMin = parseFloat(document.getElementById('amountMin').value) || 0;
            const amountMax = parseFloat(document.getElementById('amountMax').value) || Infinity;
            
            const applications = document.querySelectorAll('.application-card');
            
            applications.forEach(app => {
                const status = app.dataset.status;
                const amount = parseFloat(app.dataset.amount);
                const date = app.dataset.date;
                
                let show = true;
                
                // Status filter
                if (statusFilter && status !== statusFilter) {
                    show = false;
                }
                
                // Amount filter
                if (amount < amountMin || amount > amountMax) {
                    show = false;
                }
                
                // Date filter
                if (dateFrom && date < dateFrom) {
                    show = false;
                }
                if (dateTo && date > dateTo) {
                    show = false;
                }
                
                app.style.display = show ? 'block' : 'none';
            });
        }
        
        // Add event listeners to filters
        document.getElementById('statusFilter').addEventListener('change', filterApplications);
        document.getElementById('dateFrom').addEventListener('change', filterApplications);
        document.getElementById('dateTo').addEventListener('change', filterApplications);
        document.getElementById('amountMin').addEventListener('input', filterApplications);
        document.getElementById('amountMax').addEventListener('input', filterApplications);
        
        // Application actions
        function viewApplication(appId) {
            alert(`Viewing application details for ${appId}. In a real application, this would open a detailed view or modal.`);
        }
        
        function editApplication(appId) {
            alert(`Editing application ${appId}. In a real application, this would redirect to an edit form.`);
        }
        
        function downloadApplication(appId) {
            alert(`Downloading application ${appId} as PDF. In a real application, this would generate and download a PDF file.`);
        }
        
        function viewLoan(appId) {
            // Redirect to active loans page
            window.location.href = 'active loans.html';
        }
        
        // Set default date filter to show last 30 days
        document.addEventListener('DOMContentLoaded', function() {
            const today = new Date();
            const thirtyDaysAgo = new Date(today.getTime() - (30 * 24 * 60 * 60 * 1000));
            
            document.getElementById('dateTo').value = today.toISOString().split('T')[0];
            document.getElementById('dateFrom').value = thirtyDaysAgo.toISOString().split('T')[0];
        });

        // Toggle payment schedule visibility
        function toggleSchedule(scheduleId) {
            const table = document.getElementById(scheduleId);
            const button = table.previousElementSibling.querySelector('.schedule-toggle');
            const icon = button.querySelector('i');
            
            if (table.classList.contains('show')) {
                table.classList.remove('show');
                icon.className = 'fas fa-chevron-down';
                button.innerHTML = '<i class="fas fa-chevron-down"></i> View Schedule';
            } else {
                table.classList.add('show');
                icon.className = 'fas fa-chevron-up';
                button.innerHTML = '<i class="fas fa-chevron-up"></i> Hide Schedule';
            }
        }

        // Auto-update progress bars with animation
        function animateProgressBars() {
            const progressBars = document.querySelectorAll('.progress-fill');
            progressBars.forEach(bar => {
                const width = bar.style.width;
                bar.style.width = '0%';
                setTimeout(() => {
                    bar.style.width = width;
                }, 100);
            });
        }

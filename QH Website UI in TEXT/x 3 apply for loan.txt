        // Simple calculator for loan repayment
        document.addEventListener('DOMContentLoaded', function() {
            const loanAmountInput = document.getElementById('loanAmount');
            const loanTermInput = document.getElementById('loanTerm');
            const termTypeSelect = document.getElementById('termType');
            const calcAmount = document.getElementById('calcAmount');
            const calcTerm = document.getElementById('calcTerm');
            const calcMonthly = document.getElementById('calcMonthly');
            const calcTotal = document.getElementById('calcTotal');
            const progressFill = document.getElementById('progressFill');
            
            function updateCalculator() {
                const amount = parseFloat(loanAmountInput.value) || 0;
                let term = parseFloat(loanTermInput.value) || 0;
                const termType = termTypeSelect.value;
                
                // Convert years to months if needed
                if (termType === 'Years') {
                    term = term * 12;
                }
                
                calcAmount.textContent = '$' + amount.toLocaleString();
                calcTerm.textContent = term + ' months';
                
                if (amount > 0 && term > 0) {
                    // Since it's interest-free, monthly payment is simply amount divided by term
                    const monthlyPayment = amount / term;
                    calcMonthly.textContent = '$' + monthlyPayment.toFixed(2);
                    calcTotal.textContent = '$' + amount.toLocaleString();
                    
                    // Update progress bar (simulated)
                    const progressPercent = Math.min(100, Math.max(0, (amount / 10000) * 100));
                    progressFill.style.width = progressPercent + '%';
                } else {
                    calcMonthly.textContent = '$0';
                    calcTotal.textContent = '$0';
                    progressFill.style.width = '0%';
                }
            }
            
            // Add event listeners
            loanAmountInput.addEventListener('input', updateCalculator);
            loanTermInput.addEventListener('input', updateCalculator);
            termTypeSelect.addEventListener('change', updateCalculator);
            
            // Form submission
            const loanForm = document.getElementById('loanForm');
            const submitButton = document.getElementById('submitButton');
            const spinner = document.getElementById('spinner');
            const buttonText = document.getElementById('buttonText');
            const successMessage = document.getElementById('successMessage');
            
            loanForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                // Show loading state
                spinner.style.display = 'inline-block';
                buttonText.textContent = 'Processing...';
                submitButton.disabled = true;
                
                // Simulate API call
                setTimeout(function() {
                    // Hide loading state
                    spinner.style.display = 'none';
                    buttonText.textContent = 'Submit Application';
                    submitButton.disabled = false;
                    
                    // Show success message
                    successMessage.style.display = 'block';
                    
                    // Scroll to success message
                    successMessage.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
                }, 2000);
            });
            
            // Purpose selection
            const purposeOptions = document.querySelectorAll('.purpose-option');
            purposeOptions.forEach(option => {
                option.addEventListener('click', function() {
                    // Remove active class from all options
                    purposeOptions.forEach(opt => opt.classList.remove('active'));
                    
                    // Add active class to clicked option
                    this.classList.add('active');
                    
                    // Check the radio button
                    const radio = this.querySelector('input[type="radio"]');
                    if (radio) {
                        radio.checked = true;
                    }
                });
            });
            
            // Set initial values
            updateCalculator();
        });

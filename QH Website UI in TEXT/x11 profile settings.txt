        function handleProfilePicture(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const img = document.createElement('img');
                    img.src = e.target.result;
                    img.style.width = '100%';
                    img.style.height = '100%';
                    img.style.objectFit = 'cover';
                    
                    const avatar = document.querySelector('.profile-picture-large');
                    avatar.innerHTML = '';
                    avatar.appendChild(img);
                    
                    // Re-add the upload button
                    const uploadBtn = document.createElement('div');
                    uploadBtn.className = 'picture-upload';
                    uploadBtn.innerHTML = '<input type="file" accept="image/*" onchange="handleProfilePicture(event)"><i class="fas fa-camera"></i>';
                    avatar.appendChild(uploadBtn);
                };
                reader.readAsDataURL(file);
            }
        }
